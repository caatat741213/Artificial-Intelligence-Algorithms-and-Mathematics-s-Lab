# ==========================================
# systems1:
# 2x1 + 3x2 - 4x3 = 6
# 1x1 - 4x2 + 0x3 = 8
# ==========================================
A1 = np.array([[2, 3, -4], [1, -4, 0]])
B1 = np.array([6, 8])

rank1 = np.linalg.matrix_rank(A1)
print(f"systems1 Rank is : {rank1}，but 3 variables, so there are infinitely many solutions")
